User. 기본키(고유번호), 아이디, 비밀번호, 이름, 이메일 

create table user(
	id VARCHAR(20) PRIMARY KEY NOT NULL, #아이디(기본키)
	pwd VARCHAR(40) NOT NULL, #비밀번호
	name VARCHAR(20) NOT NULL, #이름
	email VARCHAR(40) NOT NULL, #이메일
	addr VARCHAR(100) NOT NULL #주소
)


Board. 기본키(보드넘버), 제목, 본문, 작성자(참조키), 날짜, 좋아요 갯수
	no INT PRIMARY KEY AUTO_INCERMENT, 
	title VARCHAR(100) NOT NULL, #제목
	content TEXT NOT NULL, #본문
	write VARCHAR(20) NOT NULL, #작성자
	CONSTRAINT FOREIGN KEY ('writer') REPERENCES 'User'('id')
	writedate datetime(6), #작성날짜
	liked INT DEFAULT NULL #좋아요 갯수
)

Hibernate: 
    
    create table board (
       no bigint not null,
        content varchar(255),
        liked integer not null,
        title varchar(255),
        writedate datetime,
        writer varchar(255),
        primary key (no)
    ) engine=MyISAM
